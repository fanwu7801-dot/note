## C语言如何实现继承
> C语言本身并不支持面向对象编程（OOP）中的继承概念，但我们可以通过结构体嵌套和函数指针来模拟继承的行为。
> 例如汽车有品牌、颜色等属性，电动车在汽车的基础上增加了电池容量属性。我们可以通过结构体嵌套来实现这种关系。
核心思想：把父类作为子类的第一个成员
``` cpp
typedef struct {
    char brand[50];// 品牌
    char color[20];// 颜色
    int speed;    // 速度
} Vehicle;
//定义子类
typedef struct {
    Vehicle base; // 继承自Vehicle
    int batteryCapacity; // 电池容量
} ElectricVehicle;
``` 
*注意为什么把父类作为子类的第一个成员*？
>因为C语言中，结构体的内存布局是按成员顺序排列的。 通过将父类作为子类的第一个成员，我们可以确保子类对象的内存布局与父类对象的内存布局兼容。 这样，我们可以通过类型转换将子类对象的指针转换为父类对象的指针，从而实现类似继承的行为。
![alt text](embedded-development-note-main/OOP_For_C/image-5.png)
当指针进行强转的时候，子类对象的地址和父类对象的地址是相同的，这样就可以通过父类的指针访问子类的成员,也就是实现了内存兼容
``` cpp
typedef struct {
    char brand[50];// 品牌
    char color[20];// 颜色
    int speed;    // 速度
    //函数指针 相当于方法 + 虚函数表，可以实现多态
    void (*drive)(struct Vehicle*self); // 驾驶方法
    void (*displayInfo)(struct Vehicle*self); // 显示信息方法
} Vehicle;
//定义子类
typedef struct {
    Vehicle base; // 继承自Vehicle
    int batteryCapacity; // 电池容量
} ElectricVehicle;
``` 
也就是父类定义类遥控器的按钮，子类实现具体的功能，这一点在linux内核驱动模型中有大量应用
### 动物类的例子
``` cpp
typedef struct Animal {
    char name [50];
    int age;
    void (*makeSound)(struct Animal* self); // 发出声音的方法
}Animal;

// 猫类，继承自动物类
typedef struct Cat {
    Animal base; // 继承自Animal
    int lives;   // 猫的生命值 九条命
}Cat;
// 狗类，继承自动物类
typedef struct Dog {
    Animal base; // 继承自Animal
    char breed[50]; // 狗的品种
}Dog;
// 猫发出声音的方法实现
void catSound(Animal* self) {
    printf("%s says: Meow!\n", self->name);
}
// 狗发出声音的方法实现
void dogSound(Animal* self) {
    printf("%s says: Woof!\n", self->name);
}

int main() {
    // 创建一个猫对象
    Cat* myCat = (Cat*)malloc(sizeof(Cat));
    strcpy(myCat->base.name, "Whiskers");
    myCat->base.age = 3;
    myCat->base.makeSound = catSound; // 设置发声方法
    myCat->lives = 9;

    // 创建一个狗对象
    Dog* myDog = (Dog*)malloc(sizeof(Dog));
    strcpy(myDog->base.name, "Buddy");
    myDog->base.age = 5;
    myDog->base.makeSound = dogSound; // 设置发声方法
    strcpy(myDog->breed, "Golden Retriever");

    // 使用动物类的指针调用发声方法
    Animal* animal1 = (Animal*)myCat;
    Animal* animal2 = (Animal*)myDog;

    animal1->makeSound(animal1); // 输出: Whiskers says: Meow!
    animal2->makeSound(animal2); // 输出: Buddy says: Woof!

    // 释放内存
    free(myCat);
    free(myDog);
    return 0;
}
```
小坑: 
- 记得给子类对象分配内存时，要使用子类的大小（sizeof(Cat)或sizeof(Dog)），而不是父类的大小（sizeof(Animal)）。否则会导致内存不足，可能引发未定义行为。
- 函数指针的类型必须与结构体定义中的类型匹配，否则会导致编译错误或运行时错误。
- 内存对齐

### 多态
多态(polymorphism)是面向对象编程中的一个重要概念，指的是同一操作作用于不同的对象时，可以产生不同的行为。在C语言中，我们可以通过函数指针和结构体嵌套来实现多态。
例如上面的动物类例子，不同的动物（猫和狗）都有自己的发声方法，通过函数指针实现了多态行为。
*传统的C语言困境*
>在传统的C语言中，实现多态通常依赖于条件语句（如switch或if-else）来区分不同的对象类型，并调用相应的函数。这种方式不仅代码冗长，而且难以维护和扩展。

- 代码需要对拓展开放，对修改封闭，同一个接口，不同的实现

>*函数指针*：可以指向不同的函数，实现动态绑定。

``` cpp
//虚函数表
typedef struct  {
    void (*makeSound)(struct Animal* self);
    void(*move)(struct Animal* self);
    void (*displayInfo)(struct Animal* self);
}Vtable;

//带虚函数表的动物类 基于对齐原则先写虚函数表指针
typedef struct Animal {
    Vtable* vtable; // 虚函数表指针
    char name [50];
    int age;
}Animal;

//Dog类的虚函数表
static const Vtalbe Dog_vtable = {
    .makeSound = dogSound,
    .move = dogMove,
    .displayInfo = dogDisplayInfo
};

//cat类的虚函数表
static const Vtalbe Cat_vtable = {
    .makeSound = catSound,
    .move = catMove,
    .displayInfo = catDisplayInfo
};

// 使用宏简化调用
#define ANIMAL_MAKE_SOUND(animal) ((animal)->vtable->makeSound(animal) )

//优雅调用
int main() {
    // 创建一个猫对象
    Cat* myCat = (Cat*)malloc(sizeof(Cat));
    myCat->base.vtable = &Cat_vtable; // 设置虚函数表指针
    strcpy(myCat->base.name, "Whiskers");
    myCat->base.age = 3;
    myCat->lives = 9;

    // 创建一个狗对象
    Dog* myDog = (Dog*)malloc(sizeof(Dog));
    myDog->base.vtable = &Dog_vtable; // 设置虚函数表指针
    strcpy(myDog->base.name, "Buddy");
    myDog->base.age = 5;
    strcpy(myDog->breed, "Golden Retriever");

    // 使用动物类的指针调用发声方法
    Animal* animal1 = (Animal*)myCat;
    Animal* animal2 = (Animal*)myDog;

    ANIMAL_MAKE_SOUND(animal1); // 输出: Whiskers says: Meow!
    ANIMAL_MAKE_SOUND(animal2); // 输出: Buddy says: Woof!
}
```
